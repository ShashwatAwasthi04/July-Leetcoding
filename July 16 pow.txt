class Solution {
    public double getPow(double x, long n) {
        if (n == 0) return 1.0;
        return n % 2 == 0 ? getPow(x * x, n / 2) : x * getPow(x * x, n / 2);
    }
   
    public double myPow(double x, int n) {
        long m = n;
        return m > 0 ? getPow(x, m) : 1.0 / getPow(x, -m);
    }
}
